<?php

namespace App\Modules\Blog\UI\API\Tests;

use App\Modules\Blog\Models\Comment;
use App\Ship\Abstracts\Tests\TestCase;

/**
 * @group blog
 * @group api
 */
class ListCommentsTest extends TestCase
{
    /**
     * Roles and permissions, to be attached on the user by default
     */
    protected array $access = [
        'permissions' => 'view-comment',
        'roles'       => '',
    ];

    public function testListComments(): void
    {
        $this->getTestingUser();

        Comment::factory()->count(2)->create();

        $this->getJson(route('api.comments.list'))
            ->assertOk()
            ->assertJsonStructure([
                'links',
                'meta',
                'data'
            ])
            ->assertJsonCount(Comment::query()->count(), 'data');
    }

    public function testListCommentsWithoutAccess(): void
    {
        $this->getTestingUserWithoutAccess();

        Comment::factory()->count(2)->create();

        $this->getJson(route('api.comments.list'))
            ->assertForbidden();
    }
}

<?php

namespace App\Modules\Blog\UI\API\Tests;

use App\Modules\Blog\Models\PostComment;
use App\Ship\Abstracts\Tests\TestCase;
use Illuminate\Testing\Fluent\AssertableJson;

/**
 * @group blog
 * @group api
 */
class CreatePostCommentTest extends TestCase
{
    /**
     * Roles and permissions, to be attached on the user by default
     */
    protected array $access = [
        'permissions' => 'create-post-comment',
        'roles'       => '',
    ];

    protected function getTestData(): array
    {
        return [
            // TODO: add fields here
        ];
    }

    public function testCreatePostComment(): void
    {
        $this->getTestingUser();
        $data = $this->getTestData();

        $this->postJson(route('api.post_comments.create'), $data)
            ->assertCreated()
            ->assertJson(fn (AssertableJson $json) =>
                $json->has('data', fn (AssertableJson $json) =>
                    $json->has('id')
                        ->whereAll($data)
                        ->etc()
                    )
            );

        $this->assertExistsModelWithAttributes(PostComment::class, $data);
    }

    public function testCreatePostCommentWithoutAccess(): void
    {
        $this->getTestingUserWithoutAccess();
        $data = $this->getTestData();

        $this->postJson(route('api.post_comments.create'), $data)
            ->assertForbidden();
    }
}
